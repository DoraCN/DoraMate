```rust
#[test]
fn test_undo_redo() {
    let mut dataflow = Dataflow::new();
    let mut history = History::new();

    // 添加节点
    dataflow.add_node(Node::new("node1"));
    history.commit(dataflow.clone());

    // 删除节点
    dataflow.remove_node("node1");
    history.commit(dataflow.clone());

    // 撤销
    dataflow = history.undo(&dataflow).unwrap();
    assert_eq!(dataflow.nodes.len(), 1);

    // 重做
    dataflow = history.redo(&dataflow).unwrap();
    assert_eq!(dataflow.nodes.len(), 0);
}
```

```rust
// templates.rs
pub const TEMPLATES: &[&str] = &[
    "camera-yolo-display.yml",
    "microphone-whisper-llm.yml",
    "sam2-segmentation.yml",
];

pub fn load_template(name: &str) -> Result<Dataflow, Error> {
    let path = format!("templates/{}", name);
    let yaml = fs::read_to_string(path)?;
    let dora_dataflow: DoraDataflow = serde_yaml::from_str(&yaml)?;
    Ok(Dataflow::from(&dora_dataflow))
}
```

```rust
#[test]
fn test_performance_consistency() {
    let mut fps_samples = Vec::new();

    // 运行 1000 次,记录每次渲染帧率
    for _ in 0..1000 {
        let start = Instant::now();
        render_canvas(&dataflow);
        let duration = start.elapsed();

        let fps = 1000.0 / duration.as_millis() as f64;
        fps_samples.push(fps);
    }

    // 计算标准差
    let mean = fps_samples.iter().sum::<f64>() / fps_samples.len() as f64;
    let variance = fps_samples.iter()
        .map(|&x| (x - mean).powi(2))
        .sum::<f64>() / fps_samples.len() as f64;
    let std_dev = variance.sqrt();

    // 标准差应该 < 5fps
    assert!(std_dev < 5.0, "FPS 波动过大: {:.2}", std_dev);
}
```

```bash
#!/bin/bash
# fs-stability-test.sh

echo "开始文件系统稳定性测试..."

# 1. 文件损坏测试
for i in {1..1000}; do
  # 创建随机数据流
  cat > /tmp/test_$i.yml <<EOF
nodes:
  node_$i:
    id: node_$i
    path: test
EOF

  # 计算校验和
  sha256sum /tmp/test_$i.yml > /tmp/test_$i.sha256

  # 读取文件
  doramate-cli read /tmp/test_$i.yml

  # 验证校验和
  sha256sum -c /tmp/test_$i.sha256
  if [ $? -ne 0 ]; then
    echo "❌ 文件损坏!"
    exit 1
  fi
done

echo "✅ 1000 次文件操作无损坏"

# 2. 异常中断测试 (原子写入)
for i in {1..100}; do
  # 启动保存进程
  doramate-cli save /tmp/test_atomic.yml &
  SAVE_PID=$!

  # 随机中断 (模拟崩溃)
  sleep $((RANDOM % 100))ms
  kill -9 $SAVE_PID

  # 验证文件完整性
  if [ -f /tmp/test_atomic.yml ]; then
    # 文件存在,必须是完整的
    yamllint /tmp/test_atomic.yml
    if [ $? -ne 0 ]; then
      echo "❌ 原子写入失败!"
      exit 1
    fi
  fi
done

echo "✅ 100 次异常中断测试通过"

# 3. 并发访问测试
for i in {1..10}; do
  doramate-cli save /tmp/concurrent_$i.yml &
done

wait

echo "✅ 10 个并发写入成功"
```
